import { Component, Input, ChangeDetectionStrategy } from '@angular/core';

@Component({
	selector: 'dc-dropdown-submenu',
	templateUrl: './dropdown-submenu.component.html',
	styleUrls: ['./dropdown-submenu.component.scss'],
	changeDetection: ChangeDetectionStrategy.OnPush
})
export class DropdownSubmenuComponent {
	@Input() navLinks;

	@Input() baseUrl:string = '';
	@Input() emberUrl:string = '';
	@Input() teamUrl:string = '';

	constructor() { }
}

<li class="dropdown-submenu" dropdownSubmenu>
	<a class="dropdown-item" data-toggle="dropdown">{{navLinks?.name}}<i class="material-icons dropdown-submenu-icon">play_arrow</i></a>

	<ul class="dropdown-menu" dropdownSubmenuMenu>
		<li *ngFor='let navLink of navLinks?.items'>

			<!-- submenu's submenu -->
			<dc-dropdown-submenu *ngIf='navLink?.items; else linkItem'
					[navLinks]=navLink
					[baseUrl]=baseUrl
					[emberUrl]=emberUrl
				>
			</dc-dropdown-submenu>

			<!-- single link -->
			<ng-template #linkItem>
				<!-- full link given -->
				<a *ngIf='navLink.hasFullUrl; else fullLink' class="dropdown-item" target="_blank" href="{{navLink.link}}">
					{{navLink.name}}
				</a>

				<!-- link with given emberUrl -->
				<ng-template #fullLink>
					<a 
						*ngIf='navLink.isEmber; else teamUrlLink' 
						class="dropdown-item" target="_blank" 
						href="{{emberUrl}}/{{navLink.link}}">
						{{navLink.name}}
					</a>
					
					<!-- link given teamUrl -->
					<ng-template #teamUrlLink>
						<a 
							*ngIf='navLink.isTeam; else baseUrlLink' 
							class="dropdown-item" target="_blank" 
							href="{{teamUrl}}/{{navLink.link}}">
							{{navLink.name}}
						</a>
						
						<!-- link given baseUrl -->
						<ng-template #baseUrlLink>
							<a class="dropdown-item" target="_blank" href="{{baseUrl}}/{{navLink.link}}">
								{{navLink.name}}
							</a>
						</ng-template>
					</ng-template>
				</ng-template>
			</ng-template>
		</li>
	</ul>
</li>
