import { Component, ViewChild, ViewEncapsulation, Input, ChangeDetectionStrategy} from '@angular/core';
import { ModalComponent } from '@modal';

@Component({
	selector: 'crucible-comments-modal',
	templateUrl: './crucible-comments-modal.component.html',
	styleUrls: ['./crucible-comments-modal.component.scss'],
	encapsulation: ViewEncapsulation.None,
	changeDetection: ChangeDetectionStrategy.OnPush
})
export class CrucibleCommentsModalComponent {
	modalSize = {
        width: '900px',
        height: () => window.innerHeight/1.3
    };

	@ViewChild(ModalComponent) modal: ModalComponent;
	@Input() crucibleId;
}

<dev-center-modal [modalSize]=modalSize>
	<div class='modal-header'>Crucible Comments for {{crucibleId}}</div>

	<span class='modal-body'>
		<dc-crucible-comments [crucibleId]=crucibleId></dc-crucible-comments>
	</span>

	<span class='modal-footer'>
		<button type="button" class="btn btn-outline-danger" (click)="modal.closeModal()"> Close</button>
	</span>
	
</dev-center-modal>
